\text{	Energy of mass } M, \quad E&=\sqrt{P^{2} c^{2}+M^{2} c^{4}}\\
\text{	Momentum of a mass } m,&=\sqrt{P_{1}^{2}+P_{2}^{2}}=\sqrt{P_{1}^{2}+\frac{P^{2}}{4}}\\

	\end{align*}
